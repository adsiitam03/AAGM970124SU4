40      private Node build() {
41  	LOG.info("Método build()");
42  	LOG.info("Crea 6 nodos con lestras de A hasta F");
43          /*Método que crea 6 Nodos, cada uno con una letra diferente y
44              luego a cada uno en su propiedad next le asigna en siguiente.  */
45          Node n1 = new Node("A");
46          Node n2 = new Node("B");
47          Node n3 = new Node("C");
48          Node n4 = new Node("D");
49          Node n5 = new Node("E");
50          Node n6 = new Node("F");
51          n1.next = n2;
52          n2.next = n3;
53          n3.next = n4;
54          n4.next = n5;
55          n5.next = n6;
56          return n1;
57      }
